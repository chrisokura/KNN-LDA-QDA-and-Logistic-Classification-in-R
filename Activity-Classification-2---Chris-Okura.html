knn_mod <- nearest_neighbor(neighbors = 5) %>%
  set_engine("kknn") %>%
  set_mode("classification")

logit_mod <- logistic_reg() %>%
  set_mode("classification") %>%
  set_engine("glm")

Model Recipes

ins_rec_1 <- recipe(smoker ~ charges + age + bmi, data = ins)
ins_rec_2 <- recipe(smoker ~ charges + age + bmi + sex, data = ins)
